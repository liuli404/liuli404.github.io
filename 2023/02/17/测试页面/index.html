二、环境初始化

服务器配置

系统：CentOS 7.6.1810 
IP：10.11.109.201
CPU：8线程
内存：16G
硬盘：200G
# 关闭防火墙、Selinux
systemctl stop firewalld.service && systemctl disable firewalld.service
systemctl stop NetworkManager && systemctl disable NetworkManager
setenforce 0 && sed -i 's/enforcing/disabled/g' /etc/selinux/config

# 配置 YUM 源
mv /etc/yum.repos.d/CentOS-* /tmp/
curl -o /etc/yum.repos.d/CentOS-Base.repo https://mirrors.aliyun.com/repo/Centos-7.repo
curl -o /etc/yum.repos.d/epel.repo http://mirrors.aliyun.com/repo/epel-7.repo
yum clean all && yum makecache

# 安装基本工具
yum install -y vim wget lsof net-tools bash-completion lrzsz unzip
source /usr/share/bash-completion/bash_completion

# 最大打开文件数
cat >> /etc/security/limits.conf <<EOF
* soft nofile 102400
* hard nofile 204800
EOF

# 最大进程数
cat > /etc/security/limits.d/20-nproc.conf <<EOF
* soft nproc 102400
* hard nproc 204800
EOF

# 内核参数优化
cat >> /etc/sysctl.conf << EOF
net.ipv6.conf.all.disable_ipv6 = 1
net.ipv6.conf.default.disable_ipv6 = 1
net.ipv4.icmp_echo_ignore_broadcasts = 1
net.ipv4.tcp_syncookies = 1
kernel.msgmnb = 65536
kernel.msgmax = 65536
net.ipv4.tcp_max_tw_buckets = 6000
net.ipv4.tcp_sack = 1
net.ipv4.tcp_window_scaling = 1
net.ipv4.tcp_rmem = 4096 87380 4194304
net.ipv4.tcp_wmem = 4096 16384 4194304
net.core.wmem_default = 8388608
net.core.rmem_default = 8388608
net.core.rmem_max = 16777216
net.core.wmem_max = 16777216
net.ipv4.tcp_tw_recycle = 1
vm.max_map_count = 655360
EOF
sysctl -p

MySQL

卸载自带的 mariadb

yum remove -y  mariadb* && rm -f /etc/my.cnf

创建 mysql 用户

useradd -r -s /sbin/nologin mysql

解压源码包

tar -zxvf mysql-boost-5.7.35.tar.gz

进入解压后的目录

cd mysql-5.7.35

配置

cmake . \
-DCMAKE_INSTALL_PREFIX=/data/mysql \
-DMYSQL_DATADIR=/data/mysql/data \
-DSYSCONFDIR=/data/mysql \
-DMYSQL_TCP_PORT=3306 \
-DMYSQL_UNIX_ADDR=/data/mysql/mysql.sock \
-DWITH_INNOBASE_STORAGE_ENGINE=1 \
-DWITH_PARTITION_STORAGE_ENGINE=1 \
-DWITH_FEDERATED_STORAGE_ENGINE=1 \
-DWITH_BLACKHOLE_STORAGE_ENGINE=1 \
-DWITH_MYISAM_STORAGE_ENGINE=1 \
-DENABLED_LOCAL_INFILE=1 \
-DEXTRA_CHARSETS=all \
-DDEFAULT_CHARSET=utf8mb4 \
-DDEFAULT_COLLATION=utf8mb4_general_ci \
-DWITH_SSL=system \
-DWITH_BOOST=boost

编译安装

make -j `cat /proc/cpuinfo | grep 'processor' | wc -l`&& make install

创建 mysql-files 目录并设置 750 权限

mkdir /data/mysql/mysql-files && chmod 750 /data/mysql/mysql-files

创建配置文件

vim /data/mysql/my.cnf
---
[mysqld]
port=3306
basedir=/data/mysql/
datadir=/data/mysql//data
socket=/data/mysql/mysql.sock

初始化数据库

chown -R mysql:mysql /data/mysql/
cd /data/mysql
# 这一步将获取 MySQL 初始密码
bin/mysqld --defaults-file=/data/mysql/my.cnf --initialize --user=mysql --basedir=/data/mysql/
bin/mysql_ssl_rsa_setup --datadir=/data/mysql/data

启动 MySQL

cp support-files/mysql.server /etc/init.d/mysqld
chmod +x /etc/init.d/mysqld
service mysqld start && chkconfig --add mysqld

配置环境变量

vim ~/.bashrc
---
# mysql
export PATH=$PATH:/data/mysql/bin/
source ~/.bashrc

修改 root 密码及权限

# mysql -u root -p
Enter password: # 这里输入初始密码
mysql> set password for root@localhost = password('123456Aa.');
mysql> grant all privileges on *.* to 'root'@'%' identified by '123456Aa.';
mysql> flush privileges;
mysql> CREATE DATABASE wordpress DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;

Apache

解压源码包

tar -zxvf apr-1.7.0.tar.gz 
tar -zxvf apr-util-1.6.1.tar.gz 
tar -zxvf httpd-2.4.53.tar.gz

进入解压后的目录

mv apr-1.7.0 httpd-2.4.53/srclib/apr
mv apr-util-1.6.1 httpd-2.4.53/srclib/apr-util

配置

cd httpd-2.4.53/
./configure \
--prefix=/data/httpd \
--enable-modules=all \
--enable-mods-shared=all \
--enable-cgi \
--enable-so \
--enable-rewrite \
--enable-ssl \
--enable-charset-lite \
--with-pcre

编译安装

make -j `cat /proc/cpuinfo | grep 'processor' | wc -l` && make install

配置环境变量

vim ~/.bashrc
---
# httpd
export PATH=$PATH:/data/httpd/bin/
source ~/.bashrc

设置启动文件

vim /data/httpd/bin/apachectl
# 添加
# chkconfig: 35 85 21
cp /data/httpd/bin/apachectl /etc/init.d/httpd
chmod +x /etc/init.d/httpd

修改配置文件

vim httpd/conf/httpd.conf
# ServerName 设置为 localhost:80
ServerName localhost:80


# <IfModule mime_module> 段插入以下内容
AddType application/x-httpd-php .php
AddType application/x-httpd-php-source .phps

# 修改首页文件名设置
DirectoryIndex index.html index.php

# 载入 php7_module 模块
LoadModule php7_module modules/libphp7.so

启动服务

service httpd start && chkconfig --add httpd
vim /data/httpd/htdocs/test.php

<?php
    phpinfo();
?>

访问测试页 http://10.11.109.201/test.php，解析出phpinfo页面表示成功
